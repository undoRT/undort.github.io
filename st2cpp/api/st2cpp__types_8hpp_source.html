201inline Time TIME_LITERAL(const char* literal)
202{
203 std::string str(literal);
204 Time value = 0;
205 char suffix[3] = {0};
206
207 // Parse numeric part
208 size_t i = 0;
209 while (i < str.size() && (isdigit(str[i]) || str[i] == '.')) {
210 i++;
211 }
212
213 double num = std::stod(str.substr(0, i));
214 strcpy(suffix, str.substr(i).c_str());
215
216 // Convert to milliseconds
217 if (strcmp(suffix, "ms") == 0) {
218 value = static_cast<Time>(num);
219 } else if (strcmp(suffix, "s") == 0) {
220 value = static_cast<Time>(num * 1000);
221 } else if (strcmp(suffix, "m") == 0) {
222 value = static_cast<Time>(num * 60 * 1000);
223 } else if (strcmp(suffix, "h") == 0) {
224 value = static_cast<Time>(num * 60 * 60 * 1000);
225 } else if (strcmp(suffix, "d") == 0) {
226 value = static_cast<Time>(num * 24 * 60 * 60 * 1000);
227 }
228
229 return value;
230}
