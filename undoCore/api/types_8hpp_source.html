123template<typename T, int Low, int High>
124class STArray
125{
126 static_assert(High >= Low, "High bound must be greater than or equal to Low bound");
127
128 static constexpr size_t SIZE = (High - Low + 1);
129 std::array<T, SIZE> data;
130
131public:
132 // Type aliases
133 using value_type = T;
134 using reference = T&;
135 using const_reference = const T&;
136 using iterator = typename std::array<T, SIZE>::iterator;
137 using const_iterator = typename std::array<T, SIZE>::const_iterator;
138
139 // Constructors
140 STArray() = default;
141
142 // Initializer list constructor (for 1D arrays)
143 STArray(std::initializer_list<T> init)
144 {
145 size_t i = 0;
146 for (const auto& val : init) {
147 if (i < SIZE) {
148 data[i++] = val;
149 }
150 }
151 }
152
153 // Element access with bounds checking
154 reference operator[](int index)
155 {
156 if (index < Low || index > High) {
157 throw std::out_of_range("STArray index out of bounds");
158 }
159 return data[index - Low];
160 }
161
162 const_reference operator[](int index) const
163 {
164 if (index < Low || index > High) {
165 throw std::out_of_range("STArray index out of bounds");
166 }
167 return data[index - Low];
168 }
169
170 // Bounds information
171 static constexpr int low() { return Low; }
172 static constexpr int high() { return High; }
173 static constexpr size_t size() { return SIZE; }
174
175 // Raw data access
176 T* data_ptr() { return data.data(); }
177 const T* data_ptr() const { return data.data(); }

205inline Time TIME_LITERAL(const char* literal)
206{
207 std::string str(literal);
208 size_t i = 0;
209 while (i < str.size() && (std::isdigit(str[i]) || str[i] == '.')) {
210 ++i;
211 }
212
213 double num = std::stod(str.substr(0, i));
214 std::string suffix = str.substr(i);
215
216 if (suffix == "ms") {
217 return static_cast<Time>(num);
218 }
219 if (suffix == "s") {
220 return static_cast<Time>(num * 1000);
221 }
222 if (suffix == "m") {
223 return static_cast<Time>(num * 60000);
224 }
225 if (suffix == "h") {
226 return static_cast<Time>(num * 3600000);
227 }
228 if (suffix == "d") {
229 return static_cast<Time>(num * 86400000);
230 }
231 return 0;
232}
